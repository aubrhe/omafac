Module Nat <: OrderedType.
  Definition t := nat.
  Definition eq (a b : t) := a=b.
  Definition lt (a b : t) := a<b.

  Lemma eq_refl : ∀ x : t, eq x x.

  Lemma eq_sym : ∀ x y : t, eq x y → eq y x.

  Lemma eq_trans : ∀ x y z : t, eq x y → eq y z → eq x z.

  Definition lt_trans : ∀ x y z : t, lt x y → lt y z → lt x z :=
    lt_trans.

  Definition lt_not_eq : ∀ x y : t, lt x y → ¬ eq x y :=
    NPeano.Nat.lt_neq.

Lists

Lemma strong_induction :
  ∀ n P,
    P 0 → (∀ n, (∀ m, m<n → P m) → P n)
    → P n.

Lemma strong_induction_list {A} :
  ∀ f (P : A → Prop), (∀ l, f l = 0 → P l) →
            (∀ l, (∀ m, f m < f l → P m) → P l)
            → ∀ l, P l.

Lemma app_prop_dec {A} :
  ∀ (u : list A) P,
  (∀ u1 u2, Decidable.decidable (P u1 u2)) →
    Decidable.decidable (∃ u1 u2, u = u1 ++ u2 ∧ P u1 u2).

A mesure is just a list homomorphism.
Definition mesure {A} mes :=
  (∀ u1 u2 : list A, mes (u1 ++ u2) = mes u1 + mes u2).
